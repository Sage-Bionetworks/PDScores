1% Computes basic gait test features.
 2% Inputs:
 3%  gait - gait accelerometry vector: gait(:,1) - time points,
 4%         gait(:,2:4) - X,Y,Z acceleration data
 5%
 6% (CC BY-SA 3.0) Max Little, 2014
 7function ft = features_bga(gait)
 8
 9% Output feature vector
10ft = NaN(1,7);
11
12% Ignore zero-length inputs
13N = size(gait,1);
14if (N == 0)
15    return;
16end
17
18% Calculate relative time
19t = gait(:,1)-gait(1,1);
20
21Tstart = 3.0;
22Tend   = 19.0;
23
24% Ignore gait tests which do not contain enough data
25if (max(t) < Tend)
26    return;
27end
28
29% Trim away start/end of test
30istart = find(t >= Tstart,1,'first');
31iend   = find(t <= Tend,1,'last');
32
33gaittrim = gait(istart(1):iend(1),2:4);
34ttrim = t(istart(1):iend(1));
35
36N = length(gaittrim);
37T = ttrim(end);
38dT = T-ttrim(1);
39
40% Orientation
41mg = mean(gaittrim);
42
43% Orientation-corrected force signals
44gaitforce = gaittrim-repmat(mg,N,1);
45
46% Scaled velocity signals
47dt = diff(ttrim);
48dt(end+1) = dt(end);
49gaitvel = cumsum(gaitforce.*repmat(dt,1,3));
50
51% Average scaled power X,Y,Z
52gaitpower = mean(sum(0.5*70*gaitvel.^2)/dT)/1e4;
53
54% Force vector magnitude signal
55gaitmag = sqrt(sum(gaitforce.^2,2));
56
57% Maximum force
58gaitpeak = quantile(gaitmag,0.95)/10;
59
60% Zero crossing events
61mg = mean(gaitmag);
62izc = (gaitmag(1:end-1) < mg) & (gaitmag(2:end) >= mg);
63tzc = ttrim(izc);
64dtzc = diff(tzc);
65zcr = median(dtzc);
66zcv = iqr(dtzc);
67
68% Non-uniform frequency features
69[F,P,prob] = lomb(ttrim,gaitmag-mg,4,1);
70
71% Peak frequency
72[Pmax,imax] = max(P);
73F0 = F(imax);
74probF0 = -log10(prob(imax));
75
76% SNR of peak frequency
77i = (F <= F0);
78mP = median(log10(P(i)));
79snrF0 = log10(Pmax)-mP;
80
81% Output gait test feature vector
82ft = [gaitpeak gaitpower zcr zcv F0 snrF0 probF0];
