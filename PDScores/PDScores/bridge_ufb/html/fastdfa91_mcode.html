1% Performs fast detrended fluctuation analysis on a nonstationary input signal to
 2% obtain an estimate for the scaling exponent.
 3%
 4% Useage:
 5% [alpha, intervals, flucts] = fastdfa(x)
 6% [alpha, intervals, flucts] = fastdfa(x, intervals)
 7% Inputs
 8%    x          - input signal: must be a row vector
 9% Optional inputs
10%    intervals  - List of sample interval widths at each scale
11%                 (If not specified, then a binary subdivision is constructed)
12%
13% Outputs:
14%    alpha      - Estimated scaling exponent
15%    intervals  - List of sample interval widths at each scale
16%    flucts     - List of fluctuation amplitudes at each scale
17%
18% (CC BY-SA 3.0) Max Little, 2006-2014.
19% If you use this code for academic publication, please cite:
20% M. Little, P. McSharry, I. Moroz, S. Roberts (2006),
21% Nonlinear, biophysically-informed speech pathology detection
22% in Proceedings of ICASSP 2006, IEEE Publishers: Toulouse, France.
23
24function [alpha, intervals, flucts] = fastdfa(x, varargin)
25
26if coder.target('MATLAB')
27    [xpts, ypts] = fastdfa_core(x, varargin{:});
28else
29    [m, n] = size(x);
30    if ~isempty(varargin)
31        [entries, dimensions] = size(varargin{1});
32        nscales = entries * dimensions;
33    else
34        elements = m * n;
35        nscales = floor(log2(elements));
36        if 2^(nscales - 1) > elements / 2.5
37            nscales = nscales - 1;
38        end
39    end
40    xpts = zeros(nscales,1);
41    ypts = zeros(nscales,1);
42    coder.ceval('fastdfa_core_nomex',coder.ref(xpts),coder.ref(ypts),x,m,n,varargin{:});
43end
44
45% Sort the intervals, and produce a log-log straight line fit
46datapts   = sortrows([xpts ypts],1);
47intervals = datapts(:,1);
48flucts    = datapts(:,2);
49
50coeffs    = polyfit(log10(xpts), log10(ypts), 1);
51alpha     = coeffs(1);
